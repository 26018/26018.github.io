``` c++
#include <stdio.h>
#define INF 65535
typedef struct node
{
	int data = INF;
 	struct node *left = NULL;
 	struct node *right = NULL;
}Tree;
//建立一颗二叉树 
void insert(Tree *&T,int num)
{
	if(T->data == INF)
	{
		T->data = num;
		return ;
	}
	if(T->data <= num)
	{
		if(T->right == NULL)
			T->right = new Tree;
		insert(T->right,num);
	}
	else
	{
		if(T->left == NULL)
			T->left = new Tree;
		insert(T->left,num);
	}		
}
//模式1:前序 模式2:中序 模式3:后序 
void Show(Tree *T,int mode)
{
	if(T != NULL)
	{
		if(mode == 1)
			printf("%d ",T->data);
		Show(T->left,mode);
		if(mode == 2)
			printf("%d ",T->data);
		Show(T->right,mode);
		if(mode == 3)
			printf("%d ",T->data);
	}
}
//交换左右子树 
void Exchange(Tree *&T)
{
	if(T != NULL)
	{
		Exchange(T->left);
		Exchange(T->right);
		
		Tree *m = T->left,*n = T->right;
		T->left = n;
		T->right = m;
	}
} 
//叶子结点的个数 
void Leaf_node(Tree *T,int &figure)
{
	if(T != NULL)
	{
		if(T->left == NULL && T->right == NULL)
			figure++;
		Leaf_node(T->left,figure);
		Leaf_node(T->right,figure);
	}
}
void Des_Tree(Tree *&T)
{
	if(T != NULL)
	{
		Des_Tree(T->left);
		Des_Tree(T->right);
		delete(T);
		T=NULL;
	}
}
int main()
{
	Tree *T = new Tree;
	int num[] = {6,4,7,5,9,8,15,14,2,16};
	int figure = 0;
	for(int i=0;i<sizeof(num)/sizeof(num[0]);i++)
		insert(T,num[i]); 
	printf("前序：");
	Show(T,1);
	printf("\n中序：");	
	Show(T,2);
	printf("\n后序：");
	Show(T,3);
	printf("\n");
	Leaf_node(T,figure);
	printf("叶子结点个数：%d",figure);
	printf("\n交换左右结点\n");
	Exchange(T);
	Show(T,1);
	Des_Tree(T); 
	return 0;
}
```
