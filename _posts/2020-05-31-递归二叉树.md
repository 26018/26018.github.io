``` c++
#include <stdio.h>
#define INF 65535

typedef struct node
{
	int data = INF;
 	struct node *left = NULL;
 	struct node *right = NULL;
}Tree;

void fun(Tree *&Tr,int elem)
{
	Tree *T = Tr;
	if(T->data == INF)
	{
		T->data = elem;
		return ;
	}
	if(T->data < elem)
	{
		if(T->right == NULL)
			T->right = new Tree;
		fun(T->right,elem);
	}
	else
	{
		if(T->left == NULL)
			T->left = new Tree;
		fun(T->left,elem);
	}
}

void Disp(Tree*node)
{
    if(node != NULL)
    {
        Disp(node->left);
        printf("%3d ",node->data); 
        Disp(node->right);
    }
}

void Des_Tree(Tree *&T)
{
	if(T != NULL)
	{
		Des_Tree(T->left);
		Des_Tree(T->right);
		delete(T);
		T=NULL;
	}
}

Tree *Find(Tree *&T,int elem)
{
	if(T == NULL)    //没找到 
		return NULL;
	if(T->data == elem)  ///如果是头节点 
		return T;
	if(T->left != NULL)
		if(T->left->data == elem)
			return T;
	if(T->right != NULL) 
		if(T->right->data == elem)
			return T;	
	T->data < elem ? Find(T->right,elem) : Find(T->left,elem);
}

void getElem(Tree *&T,int num[])
{	
	if(T != NULL)
	{
		getElem(T->left,num);
		num[++num[0]] = T->data;
		getElem(T->right,num);
	}
}

void delElem(Tree *&T,int elem)
{
	Tree *p = Find(T,elem);
	Tree *ep = p;
	int num[100] = {0};
	if(p == NULL)  //如果没找到 
		return ;
	if(p->data == elem) //如果是头节点 
	{
		printf("\n找到：%3d\n",p->data);
		getElem(p,num);
		Des_Tree(p->left);
		Des_Tree(p->right);
		p->data = INF;
	}
	else
	{
		p->data < elem ? (ep = p->right,p->right = NULL): (ep = p->left,p->left = NULL);
		printf("\n找到：%3d\n",ep->data);
		getElem(ep,num);
		Des_Tree(ep);
	}
	for(int i=1;i<=num[0];i++) 
		if(num[i] != elem)
			fun(T,num[i]);
} 

int main()
{
	Tree *T = new Tree;
	int a[] = {4,8,7,3,84,26,55,48,73,25,17,3,18,100};	
	for(int i=0;i<sizeof(a)/sizeof(a[0]);i++)
		fun(T,a[i]);
		
	Disp(T);
	
	delElem(T,4);
	delElem(T,104);
	delElem(T,3);
	delElem(T,18);
	delElem(T,100);
	
	Disp(T);
	Des_Tree(T); 
	return 0;
}
```
