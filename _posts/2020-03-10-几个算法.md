- ## 删除与指定元素相同数字

主要算法：
```
typedef struct
{
	int data[10];
	int length;
}Sqlist;

void delnodel(Sqlist *&L,int x)
{
	int k = 0,i;
	for(i=0;i<L->length;i++)
		if(L->data[i]!=x)
		{
			L->data[k]=L->data[i];
			k++; 
		}
	L->length=k;
}
```
- ## 将奇数移动到偶数前面     

主要算法：
```
void swap(int &a,int &b)
{
	int temp = a;
	a= b;
	b = temp;
}

void movel(Sqlist *&L)
{
	int i=0,j=L->length-1;
	while(i<j)
	{
		while(i<j&&L->data[j]%2==0)
			j--;
		while(i<j&&L->data[i]%2==1)
			i++;
		if(i<j)
			swap(L->data[i],L->data[j]);
	}
}
```
- ## 将小于第一个数的数字移动到它前面 
 
主要算法：
```
void partitionl(Sqlist *&L)
{
	int i=0,j=L->length-1;
	int pivot = L->data[0];
	while(i<j)
	{
		while(j>i&&L->data[j]>pivot)
			j--;
		L->data[i] = L->data[j];
		while(i<j&&L->data[i]<=pivot)
			i++;
		L->data[j] = L->data[i]; 
	}
	L->data[i] = pivot;
}
```
